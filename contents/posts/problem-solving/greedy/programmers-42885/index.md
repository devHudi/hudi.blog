def solution(people, limit):
    remains = deque(sorted(people))
    count = 0

    while len(remains) > 0:
        boat = 0

        while True:
            if len(remains) == 1:
                count += 1
                remains.popleft()
                break

            if boat > 0: boat += remains.popleft()

                count += 1
                break

    return count
